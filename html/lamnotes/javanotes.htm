to execute a shell cmd:
----------------------
void doCmd(String a) throws Exception{
	Process p = Runtime.getRuntime().exec(a);
	InputStream stdOut = p.getInputStream();
	InputStream stdErr = p.getErrorStream();
	OutputStream stdIn = p.getOutputStream();
	p.waitFor();

	stdOut.close();
	stdErr.close();
	stdIn.close();
    }
