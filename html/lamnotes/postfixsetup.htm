Email Server Setup

L. Gerke - 1/2003

(Successful with Red Hat 7.3).





mv /usr/sbin/sendmail /usr/sbin/sendmail.OFF
mv /usr/bin/newaliases /usr/bin/newaliases.OFF
mv /usr/bin/mailq /usr/bin/mailq.OFF
chmod 755 /usr/sbin/sendmail.OFF /usr/bin/newaliases.OFF /usr/bin/mailq.OFF
/usr/sbin/adduser -M postfix

add the following to /etc/aliases:
# postfix install
postfix:	root

/usr/sbin/groupadd postdrop

make;
make install
[defaults]


add to /etc/postfix/main.cf:
myhostname = gerke-oshea.ucsf.edu
mydestination = $myhostname, localhost.$mydomain
mynetworks_style = host


add to /etc/aliases
# adding during postfix install
root:		lcgerke


postfix start
[I HAD TO STOP THE SENDMAIL SERVICE TO MAKE THIS WORK]
